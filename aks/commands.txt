Install the Azure CLI
	https://docs.microsoft.com/en-us/cli/azure/install-azure-cli
	https://docs.microsoft.com/en-us/cli/azure/install-azure-cli-windows?tabs=azure-cli
--
az version

{
  "azure-cli": "2.16.0",
  "azure-cli-core": "2.16.0",
  "azure-cli-telemetry": "1.0.6",
  "extensions": {}
}
--
az login
--
Create a resource group
az group create --name myResourceGroup --location westeurope
--
Create an Azure Container Registry
az acr create --resource-group myResourceGroup --name shoppingamitacr --sku Basic
--
Enable Admin Account for ACR Pull
az acr update -n shoppingamitacr --admin-enabled true
--
Log in to the container registry
az acr login --name shoppingamitacr
Login Succeeded
--
Tag a container image

get the login server address
az acr list --resource-group myResourceGroup --query "[].{acrLoginServer:loginServer}" --output table
shoppingamitacr.azurecr.io
--
Tag your images

docker tag shoppingapi:latest shoppingamitacr.azurecr.io/shoppingapi:v1
docker tag shoppingclient:latest shoppingamitacr.azurecr.io/shoppingclient:v1

Check
docker images
--
Push images to registry

docker push shoppingamitacr.azurecr.io/shoppingapi:v1
docker push shoppingamitacr.azurecr.io/shoppingclient:v1
--
List images in registry
az acr repository list --name shoppingamitacr --output table

Result
shoppingapi
shoppingclient
--
See tags
az acr repository show-tags --name shoppingamitacr --repository shoppingclient --output table

Result
v1
--
Create AKS cluster with attaching ACR
az aks create --resource-group myResourceGroup --name myAKSCluster --node-count 1 --generate-ssh-keys --attach-acr shoppingamitacr

--
Install the Kubernetes CLI
az aks install-cli
--
Connect to cluster using kubectl
az aks get-credentials --resource-group myResourceGroup --name myAKSCluster

To verify;
kubectl get nodes
kubectl get all
--
Check Kube Config

kubectl config get-contexts
kubectl config current-context
kubectl config use-context gcpcluster-k8s-1
	Switched to context "gcpcluster-k8s-1"
--
Deploy microservices to AKS

kubectl apply -f .\aks\
--
Clean All AKS and Azure Resources

az group delete --name myResourceGroup --yes --no-wait
